  ollama:

      - "11434:11434"
      - "3000:3000"
    volumes:
      - ollama_data:/root/.ollama
      - ./config:/app/config
    environment:
      - OLLAMA_MODELS=llama2,mistral
      - ENABLE_CONTENT_AGENT=true
